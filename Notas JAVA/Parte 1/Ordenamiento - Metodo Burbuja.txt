public static void burbuja(int[]vector){
        int temp;
        for(int i=1;i < vector.length;i++){
            for (int j=0 ; j < vector.length- 1; j++){
                if (vector[j] > vector[j+1]){
                    temp = vector[j];
                    vector[j] = vector[j+1];
                    vector[j+1] = temp;
                }
            }
        }
    }
